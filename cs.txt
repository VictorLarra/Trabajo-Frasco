
 public class FrascoDulce {
    int capacidadMaxima = 200;
    int contenidoActual = 0;


    public void sacar(int cantidad) {
        if (cantidad <= contenidoActual) {
            contenidoActual -= cantidad;
            System.out.println("Se sacaron " + cantidad + " gramos.");
        } else {
            System.out.println("No hay suficiente dulce.");
        }
    }

    public void mostrarContenido() {
        System.out.println("El frasco contiene actualmente " + contenidoActual + " gramos.");
    }

    public static void main(String[] args) {
        FrascoDulce frasco = new FrascoDulce();
        frasco.envasar(250);
        frasco.sacar(150);
        frasco.mostrarContenido();
    }
}
